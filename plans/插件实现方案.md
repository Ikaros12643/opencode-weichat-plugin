## 最终实现方案

### 一、工作流程

```
微信消息 → WeChat 插件 → 存储上下文 → SDK 调用 → 专用会话
                                              ↓
                                        AI 自动回复
                                              ↓
                                    wechat_reply 工具
                                              ↓
                                         发送到微信
```


#### `src/index.ts` - 插件入口

```typescript
import type { Plugin } from "@opencode-ai/plugin"
import { tool } from "@opencode-ai/plugin"
import { WeChatClient } from "./client"
import { WeChatStore } from "./store"
import { formatMessage } from "./message"

export const WeChatPlugin: Plugin = async (input) => {
  const { client, directory } = input
  const store = new WeChatStore(directory)
  const wechat = new WeChatClient(store)
  
  // 初始化连接
  await wechat.init()
  
  // 启动轮询
  wechat.startPolling(async (msg) => {
    // 获取专用会话
    let sessionID = store.getSessionID()
    if (!sessionID) {
      const res = await client.session.create({ title: "WeChat" })
      sessionID = res.data.id
      store.setSessionID(sessionID)
    }
    
    // 存储上下文 token
    store.cacheContext(msg.senderId, msg.contextToken)
    
    // 注入消息到会话
    await client.session.promptAsync({
      sessionID,
      parts: [{ type: "text", text: formatMessage(msg) }],
    })
  })
  
  return {
    tool: {
      wechat_reply: tool({
        description: "回复微信消息（纯文本，无markdown）",
        args: {
          sender_id: tool.schema.string(),
          text: tool.schema.string(),
        },
        async execute(args) {
          await wechat.sendText(args.sender_id, args.text)
          return "已发送"
        },
      }),
      
      wechat_send_image: tool({
        description: "发送微信图片",
        args: {
          sender_id: tool.schema.string(),
          file_path: tool.schema.string(),
        },
        async execute(args) {
          await wechat.sendImage(args.sender_id, args.file_path)
          return "已发送图片"
        },
      }),
    },
  }
}

export default WeChatPlugin
```

#### `src/store.ts` - 存储管理

```typescript
import fs from "node:fs"
import path from "node:path"

export class WeChatStore {
  private dir: string
  private ctxTokens = new Map<string, string>()
  
  constructor(projectDir: string) {
    this.dir = path.join(process.env.HOME || "~", ".opencode", "wechat")
    fs.mkdirSync(this.dir, { recursive: true })
    this.loadContextTokens()
  }
  
  loadCredentials() {
    try {
      const file = path.join(this.dir, "account.json")
      if (!fs.existsSync(file)) return null
      return JSON.parse(fs.readFileSync(file, "utf-8"))
    } catch { return null }
  }
  
  saveCredentials(data: any) {
    fs.writeFileSync(
      path.join(this.dir, "account.json"),
      JSON.stringify(data, null, 2)
    )
  }
  
  getSessionID(): string | null {
    try {
      const file = path.join(this.dir, "session.json")
      if (!fs.existsSync(file)) return null
      return JSON.parse(fs.readFileSync(file, "utf-8")).sessionID
    } catch { return null }
  }
  
  setSessionID(id: string) {
    fs.writeFileSync(
      path.join(this.dir, "session.json"),
      JSON.stringify({ sessionID: id })
    )
  }
  
  cacheContext(senderId: string, token: string) {
    this.ctxTokens.set(senderId, token)
    fs.writeFileSync(
      path.join(this.dir, "context.json"),
      JSON.stringify(Object.fromEntries(this.ctxTokens))
    )
  }
  
  getContext(senderId: string): string | null {
    return this.ctxTokens.get(senderId) ?? null
  }
  
  private loadContextTokens() {
    try {
      const file = path.join(this.dir, "context.json")
      if (fs.existsSync(file)) {
        const data = JSON.parse(fs.readFileSync(file, "utf-8"))
        this.ctxTokens = new Map(Object.entries(data))
      }
    } catch {}
  }
}
```

#### `src/message.ts` - 消息格式化

```typescript
export function formatMessage(msg: any): string {
  const content = extractContent(msg)
  const sender = msg.from_user_id?.split("@")[0] ?? "未知"
  const canReply = msg.context_token ? "true" : "false"
  
  return `<wechat_message sender="${sender}" sender_id="${msg.from_user_id}" can_reply="${canReply}">
${content}

请使用 wechat_reply 工具回复。如果 can_reply="false"，告诉用户发送下一条消息以建立会话。
</wechat_message>`
}

function extractContent(msg: any): string {
  for (const item of msg.item_list ?? []) {
    switch (item.type) {
      case 1: return item.text_item?.text ?? ""
      case 2: return `[图片]`
      case 3: return `[语音] ${item.voice_item?.text ?? ""}`
      case 4: return `[文件] ${item.file_item?.file_name ?? ""}`
      case 5: return `[视频]`
    }
  }
  return "[未知消息类型]"
}
```
